# 我的日常功能实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在添加页为母亲膳食管家增加“我的日常”快捷选食物入口，自动汇总最近 30 天选过的食物并预填上次数量/单位。

**Architecture:** 不新增 IndexedDB 表，由 `DataService.getRecentFoods()` 从 `daily_intakes` 实时计算；`add.html` 步骤从 4 步改为 3 步，step2 默认展示“我的日常”，可切换为分类查找；`src/common.js` 新增相对时间辅助函数用于 UI 展示。

**Tech Stack:** 原生 HTML/JS/CSS，IndexedDB，无构建步骤，经典 `<script defer>` 全局模块。

## Global Constraints

- 不新增 IndexedDB object store，不升级 `DB_VERSION`。
- “我的日常”时间窗口固定为最近 30 天，按最近使用时间倒序，同一食物去重并保留最新一次的数量/单位。
- 从“我的日常”选择食物时预填上次数量/单位；单位已不存在时回退到当前食物的默认单位或 `g`。
- 所有动态 HTML 必须使用 `escapeHtml()` 转义。
- 代码变更需伴随 `tests/test.html` 新增用例，所有测试通过后再进入下一任务。
- 遵循现有全局挂载约定：`window.App`、`window.DataService`。

```js
App.formatRelativeDate = function formatRelativeDate(isoString) {
  if (!isoString) return '';
  const date = new Date(isoString);
  if (Number.isNaN(date.getTime())) return '';

  const today = new Date();
  const dateStr = App.formatLocalDate(date);
  const todayStr = App.formatLocalDate(today);
  if (dateStr === todayStr) return '今天';

  const yesterday = new Date(today);
  yesterday.setDate(today.getDate() - 1);
  if (dateStr === App.formatLocalDate(yesterday)) return '昨天';

  const diffMs = today.setHours(0, 0, 0, 0) - new Date(dateStr).setHours(0, 0, 0, 0);
  const diffDays = Math.floor(diffMs / 86400000);
  if (diffDays > 1) return `${diffDays}天前`;
  return '';
};
```

```js
// ========== 我的日常 ==========
DataService.getRecentFoods = async function getRecentFoods({
  userId,
  days = 30,
  limit = null
} = {}) {
  const uid = userId || App.getCurrentUserId();
  const end = new Date();
  const start = new Date();
  start.setDate(end.getDate() - days + 1);

  const startStr = App.formatLocalDate(start);
  const endStr = App.formatLocalDate(end);

  const db = await getDB();
  let intakes;
  try {
    intakes = await db.getIntakesByDateRange(uid, startStr, endStr);
  } catch (err) {
    console.error('getRecentFoods query failed:', err);
    return [];
  }

  const latestMap = new Map();
  for (const intake of intakes) {
    const existing = latestMap.get(intake.food_id);
    if (!existing || (intake.created_at && existing.created_at && intake.created_at > existing.created_at)) {
      latestMap.set(intake.food_id, intake);
    }
  }

  const sorted = Array.from(latestMap.values()).sort((a, b) => {
    const ta = a.created_at ? new Date(a.created_at).getTime() : 0;
    const tb = b.created_at ? new Date(b.created_at).getTime() : 0;
    return tb - ta;
  });

  const results = [];
  for (const intake of sorted) {
    const food = await db.getFoodById(intake.food_id);
    if (!food) continue;
    results.push({
      food,
      lastQuantity: Number(intake.quantity) || 1,
      lastUnit: intake.unit || 'g',
      lastUnitWeightG: Number(intake.unit_weight_g) || 1,
      lastUsedAt: intake.created_at || '',
      lastUsedDate: intake.date || ''
    });
  }

  return limit ? results.slice(0, limit) : results;
};
```

```js
function showStep(n) {
  document.querySelectorAll('.step').forEach(s => {
    const stepNum = parseInt(s.dataset.step);
    s.classList.remove('active', 'done');
    if (stepNum === n) s.classList.add('active');
    else if (stepNum < n) s.classList.add('done');
  });

  for (let i = 1; i <= 3; i++) {
    const el = document.getElementById('step' + i);
    if (el) el.classList.toggle('hidden', i !== n);
  }

  currentStep = n;
  updateButtons();
}

function updateButtons() {
  const backBtn = document.getElementById('btnBack');
  const nextBtn = document.getElementById('btnNext');

  if (currentStep === 1) {
    backBtn.textContent = '取消';
    nextBtn.textContent = '下一步';
    nextBtn.disabled = !selectedMeal;
  } else if (currentStep === 2) {
    backBtn.textContent = '上一步';
    nextBtn.textContent = '下一步';
    nextBtn.disabled = true; // step2 通过点击食物直接进入 step3
  } else if (currentStep === 3) {
    backBtn.textContent = '上一步';
    nextBtn.textContent = '✓ 确认保存';
    nextBtn.disabled = false;
  }
}

async function nextStep() {
  if (currentStep === 1 && selectedMeal) {

    await switchFoodSource('recent');

}

function prevStep() {
  if (currentStep === 1) {

  }
  if (currentStep === 3) {
```

```js
}

async function selectRecentFood(foodId, lastQty, lastUnit, lastUnitWeightG) {
  selectedFood = await DataService.getFoodById(foodId);
  if (!selectedFood) return;

  currentQty = Number(lastQty) || 1;
  const units = await DataService.getFoodUnits(foodId);
  const matched = units.find(u => u.name === lastUnit);
  if (matched) {
    currentUnit = { name: matched.name, weight: matched.weight_g };
  } else if (units.length) {
    currentUnit = { name: units[0].name, weight: units[0].weight_g };
  } else {
    currentUnit = { name: 'g', weight: 1 };
  }

  showStep(3);
  updateQtyDisplay();
  updateQtyPreview();
```
